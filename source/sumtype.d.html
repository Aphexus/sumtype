4 This module provides [SumType], an alternative to `std.variant.Algebraic` with
5 [match|improved pattern-matching], full attribute correctness (`pure`, `@safe`,

7 runtime type information (`TypeInfo`).

14 /** $(H3 Structural matching)
15  *
16  * In the `length` and `horiz` functions below, the handlers for `match` do not
17  * specify the types of their arguments. Instead, matching is done based on the
18  * type's structure: any type with `x` and `y` properties will be matched by the
19  * `rect` handlers, and any type with `r` and `theta` properties will be matched
20  * by the `polar` handlers.
21  */
22 unittest {
23 	import std.math: approxEqual, cos, PI, sqrt;
24 
25 	struct Rectangular { double x, y; }
26 	struct Polar { double r, theta; }
27 	alias Vector = SumType!(Rectangular, Polar);
28 
29 	pure @safe @nogc nothrow
30 	double length(Vector v)
31 	{
32 		return v.match!(
33 			rect => sqrt(rect.x^^2 + rect.y^^2),
34 			polar => polar.r
35 		);
36 	}
37 
38 	pure @safe @nogc nothrow
39 	double horiz(Vector v)
40 	{
41 		return v.match!(
42 			rect => rect.x,
43 			polar => polar.r * cos(polar.theta)
44 		);
45 	}
46 
47 	Vector u = Rectangular(1, 1);
48 	Vector v = Polar(1, PI/4);
49 
50 	assert(length(u).approxEqual(sqrt(2.0)));
51 	assert(length(v).approxEqual(1));
52 	assert(horiz(u).approxEqual(1));
53 	assert(horiz(v).approxEqual(sqrt(0.5)));
54 }

56 /** $(H3 Arithmetic expression evaluator)
57  *
58  * This example makes use of the special placeholder type `This` to define a
59  * [https://en.wikipedia.org/wiki/Recursive_data_type|recursive data type]: an
60  * [https://en.wikipedia.org/wiki/Abstract_syntax_tree|abstract syntax tree] for
61  * representing simple arithmetic expressions.
62  */
63 unittest {
64 	import std.functional: partial;
65 	import std.traits: EnumMembers;
66 	import std.typecons: Tuple;
67 
68 	enum Op : string
69 	{
70 		Plus  = "+",
71 		Minus = "-",
72 		Times = "*",
73 		Div   = "/"
74 	}
75 
76 	// An expression is either
77 	//  - a number,
78 	//  - a variable, or
79 	//  - a binary operation combining two sub-expressions.
80 	alias Expr = SumType!(
81 		double,
82 		string,
83 		Tuple!(Op, "op", This*, "lhs", This*, "rhs")
84 	);
85 
86 	// Shorthand for a binary operation
87 	alias BinOp = Expr.Types[2];
88 
89 	// Factory for number expressions
90 	pure @safe
91 	Expr* num(double value)
92 	{
93 		return new Expr(value);
94 	}
95 
96 	// Factory for variable expressions
97 	pure @safe
98 	Expr* var(string name)
99 	{
100 		return new Expr(name);
101 	}
102 
103 	// Factory for binary operation expressions
104 	pure @safe
105 	Expr* binOp(Op op, Expr* lhs, Expr* rhs)
106 	{
107 		return new Expr(BinOp(op, lhs, rhs));
108 	}
109 
110 	// Convenience wrappers for creating BinOp expressions
111 	alias sum  = partial!(binOp, Op.Plus);
112 	alias diff = partial!(binOp, Op.Minus);
113 	alias prod = partial!(binOp, Op.Times);
114 	alias quot = partial!(binOp, Op.Div);
115 
116 	// Evaluate expr, looking up variables in env
117 	pure @safe nothrow
118 	double eval(Expr expr, double[string] env)
119 	{
120 		return expr.match!(
121 			(double num) => num,
122 			(string var) => env[var],
123 			(BinOp bop) {
124 				double lhs = eval(*bop.lhs, env);
125 				double rhs = eval(*bop.rhs, env);
126 				final switch(bop.op) {
127 					static foreach(op; EnumMembers!Op) {
128 						case op:
129 							return mixin("lhs" ~ op ~ "rhs");
130 					}
131 				}
132 			}
133 		);
134 	}
135 
136 	// Return a "pretty-printed" representation of expr
137 	@safe
138 	string pprint(Expr expr)
139 	{
140 		import std.format;
141 
142 		return expr.match!(
143 			(double num) => "%g".format(num),
144 			(string var) => var,
145 			(BinOp bop) => "(%s %s %s)".format(
146 				pprint(*bop.lhs),
147 				bop.op,
148 				pprint(*bop.rhs)
149 			)
150 		);
151 	}
152 
153 	Expr* myExpr = sum(var("a"), prod(num(2), var("b")));
154 	double[string] myEnv = ["a":3, "b":4, "c":7];
155 
156 	assert(eval(*myExpr, myEnv) == 11);
157 	assert(pprint(*myExpr) == "(a + (2 * b))");
158 }

162 /**
163  * A tagged union that can hold a single value from any of a specified set of
164  * types
165  *
166  * The stored value can be operated on using [match|pattern matching].
167  *
168  * The special type `This` can be used as a placeholder to create
169  * self-referential types, just like with `Algebraic`.
170  *
171  * See_Also: `std.variant.Algebraic`
172  */
173 struct SumType(TypeParams...)
174 {
175 	import std.meta: AliasSeq, staticIndexOf;
176 	import std.typecons: ReplaceType;
177 
178 	/// The types a `SumType` can hold
179 	alias Types = AliasSeq!(ReplaceType!(This, typeof(this), TypeParams));
180 
181 private:
182 
183 	int tag;
184 
185 	union
186 	{
187 		Types values;
188 	}
189 
190 	alias value(T) = values[staticIndexOf!(T, Types)];
191 
192 public:
193 
194 	static foreach (i, T; Types) {
195 		/// Constructs a `SumType` holding a specific value
196 		this(T val)
197 		{
198 			tag = i;
199 			value!T = val;
200 		}
201 	}
202 
203 	import std.traits: isAssignable;
204 
205 	static foreach (i, T; Types) {
206 		static if (isAssignable!T) {
207 			/// Assigns a value to a `SumType`
208 			void opAssign(T rhs)
209 			{
210 				tag = i;
211 				value!T = rhs;
212 			}
213 		}
214 	}
215 
216 	/// Returns a string representation of the currently-held value
217 	auto toString()
218 	{
219 		import std.conv: to;
220 
221 		return this.match!(value => value.to!string);
222 	}
223 }

317 /**
318  * Calls a type-appropriate function with the value held in a [SumType].
319  *
320  * For each possible type the [SumType] can hold, the given handlers are
321  * checked, in order, to see whether they accept a single argument of that type.
322  * The first one that does is chosen as the match for that type.
323  *
324  * Implicit conversions are not taken into account, except between
325  * differently-qualified versions of the same type. For example, a handler that
326  * accepts a `long` will not match the type `int`, but a handler that accepts a
327  * `const(int)[]` will match the type `immutable(int)[]`.
328  *
329  * Every type must have a matching handler. This is enforced at compile-time.
330  *
331  * Handlers may be functions, delegates, or objects with opCall overloads.
332  * Templated handlers are also accepted, and will match any type for which they
333  * can be implicitly instantiated.
334  *
335  * Returns:
336  *   The value returned from the handler that matches the currently-held type.
337  *
338  * See_Also: `std.variant.visit`
339  */
340 template match(handlers...)
341 {
342 	/**
343 	 * The actual `match` function.
344 	 *
345 	 * Params:
346 	 *   self = A [SumType] object
347 	 */
348 	auto match(Self : SumType!TypeParams, TypeParams...)(Self self)
349 	{
350 		alias Types = self.Types;
351 
352 		pure static auto getHandlerIndices()
353 		{
354 			import std.meta: staticIndexOf;
355 			import std.traits: hasMember, isCallable, isSomeFunction, Parameters;
356 
357 			// immutable overrides all other qualifiers, so this is true if and
358 			// only if the two types are the same up to qualifiers.
359 			enum sameBaseType(T, U) = is(immutable(T) == immutable(U));
360 
361 			int[Types.length] indices;
362 			indices[] = -1;
363 
364 			void setHandlerIndex(T)(int hid)
365 				if (staticIndexOf!(T, Types) >= 0)
366 			{
367 				int tid = staticIndexOf!(T, Types);
368 
369 				if (indices[tid] == -1) {
370 					indices[tid] = hid;
371 				}
372 			}
373 
374 			static foreach (T; Types) {
375 				static foreach (i, h; handlers) {
376 					static if (is(typeof(h(T.init)))) {
377 						// Regular handlers
378 						static if (isCallable!h) {
379 							// Functions and delegates
380 							static if (isSomeFunction!h) {
381 								static if (sameBaseType!(T, Parameters!h[0])) {
382 									setHandlerIndex!T(i);
383 								}
384 							// Objects with overloaded opCall
385 							} else static if (hasMember!(typeof(h), "opCall")) {
386 								static foreach (overload; __traits(getOverloads, typeof(h), "opCall")) {
387 									static if (sameBaseType!(T, Parameters!overload[0])) {
388 										setHandlerIndex!T(i);
389 									}
390 								}
391 							}
392 						// Generic handlers
393 						} else {
394 							setHandlerIndex!T(i);
395 						}
396 					}
397 				}
398 			}
399 			return indices;
400 		}
401 
402 		enum handlerIndices = getHandlerIndices;
403 
404 		final switch (self.tag) {
405 			static foreach (i, T; Types) {
406 				static assert(handlerIndices[i] != -1,
407 					"No matching handler for type `" ~ T.stringof ~ "`");
408 
409 					case i:
410 						return handlers[handlerIndices[i]](self.value!T);
411 			}
412 		}
413 		assert(false); // unreached
414 	}
415 }
